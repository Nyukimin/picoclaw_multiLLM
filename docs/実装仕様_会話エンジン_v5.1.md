## 1. 概要

v5.0 で構築した3層記憶インフラ（Redis/DuckDB/Qdrant）の上に、**ConversationEngine** を実装する。
ConversationEngine は Mio の「記憶脳」として、Recall 結果を RecallPack に変換し、LLM プロンプトに注入する。

### 1.1 実装目標

| 目標 | 説明 |
|------|------|
| Recall 結果のプロンプト注入 | 現在捨てている Recall 結果を LLM プロンプトに反映 |
| ペルソナ付き応答 | ミオのキャラクター設定をシステムプロンプトに注入 |
| ユーザープロファイル蓄積 | ユーザーの好み・事実を長期記憶として蓄積 |
| Thread 自動判定 | 話題の継続/切替を自動判定 |

### 2.1 ConversationEngine インターフェース

**ファイル**: `internal/domain/conversation/engine.go`

```go
package conversation

import "context"

// ConversationEngine は Mio の「記憶脳」
// ConversationManager をラップし、RecallPack 生成と Thread 管理を提供
type ConversationEngine interface {
    // BeginTurn はターン開始時に呼び出し、RecallPack を返す
    // 1. Recall（想起）
    // 2. Thread 判定（継続 or 新規）
    // 3. RecallPack 構築（Persona + UserProfile 含む）
    BeginTurn(ctx context.Context, sessionID string, userMessage string) (*RecallPack, error)

    // EndTurn はターン終了時に呼び出し、メッセージ保存と Thread 管理を実行
    // 1. ユーザーメッセージ + 応答の Store
    // 2. Thread 終了判定 → FlushThread
    // 3. UserProfile 更新（プロファイル抽出）
    EndTurn(ctx context.Context, sessionID string, userMessage string, response string) error

    // GetPersona は現在のペルソナ設定を返す
    GetPersona() PersonaState

    // GetUserProfile はユーザープロファイルを返す（なければ空）
    GetUserProfile(ctx context.Context, userID string) (UserProfile, error)
}
```

```go
package conversation

// RecallPack は Recall 結果を構造化した LLM プロンプト注入用フォーマット
type RecallPack struct {
    // ShortContext: 現在の Thread 内の直近メッセージ（最大12件）
    ShortContext []Message

    // MidSummaries: 同一セッション内の過去 Thread 要約（最大3件）
    MidSummaries []ThreadSummary

    // LongFacts: VectorDB から類似検索した過去の知識（最大3件）
    LongFacts []string

    // KBSnippets: ドメイン知識ベースからの関連情報（最大2件）
    KBSnippets []string

    // Persona: キャラクター設定
    Persona PersonaState

    // UserProfile: ユーザーの好み・傾向
    UserProfile UserProfile

    // Constraints: トークン上限等
    Constraints PromptConstraints
}

// PromptConstraints はプロンプト組み立ての制約
type PromptConstraints struct {
    MaxTotalTokens    int // LLM の MaxContext（デフォルト: 8192）
    MaxPromptTokens   int // プロンプトに使えるトークン（デフォルト: 4000）
    MaxResponseTokens int // 応答用トークン（デフォルト: 512）
}

// DefaultConstraints はデフォルトのトークン制約を返す
func DefaultConstraints() PromptConstraints {
    return PromptConstraints{
        MaxTotalTokens:    8192,
        MaxPromptTokens:   4000,
        MaxResponseTokens: 512,
    }
}

// ToPromptMessages は RecallPack を llm.Message のスライスに変換
// Mio.Chat() でそのまま使えるフォーマット
func (rp *RecallPack) ToPromptMessages(userMessage string) []Message {
    // プロンプト組み立てロジックは実装仕様 §3 を参照
    return nil // 実装時に詳細化
}

// HasContext は RecallPack に何らかの文脈があるかを返す
func (rp *RecallPack) HasContext() bool {
    return len(rp.ShortContext) > 0 ||
        len(rp.MidSummaries) > 0 ||
        len(rp.LongFacts) > 0 ||
        len(rp.KBSnippets) > 0
}
```

### 2.4 UserProfile 構造体

**ファイル**: `internal/domain/conversation/user_profile.go`

```go
package conversation

import "time"

// UserProfile はユーザーの好み・傾向を管理
type UserProfile struct {
    UserID      string            `json:"user_id"`
    Preferences map[string]string `json:"preferences"` // "好きな映画ジャンル": "SF"
    Facts       []string          `json:"facts"`        // "エンジニア", "Go言語を使用"
    UpdatedAt   time.Time         `json:"updated_at"`
}

// NewUserProfile は空の UserProfile を作成
func NewUserProfile(userID string) UserProfile {
    return UserProfile{
        UserID:      userID,
        Preferences: make(map[string]string),
        Facts:       []string{},
        UpdatedAt:   time.Now(),
    }
}

// Merge は新しいプロファイル情報をマージする
func (p *UserProfile) Merge(newPrefs map[string]string, newFacts []string) {
    for k, v := range newPrefs {
        p.Preferences[k] = v
    }
    // 重複を避けてFactsを追加
    existing := make(map[string]bool)
    for _, f := range p.Facts {
        existing[f] = true
    }
    for _, f := range newFacts {
        if !existing[f] {
            p.Facts = append(p.Facts, f)
        }
    }
    p.UpdatedAt = time.Now()
}

// ToPromptText はプロンプト注入用のテキストを返す
func (p *UserProfile) ToPromptText() string {
    if len(p.Preferences) == 0 && len(p.Facts) == 0 {
        return ""
    }
    text := "【ユーザーについて知っていること】\n"
    for k, v := range p.Preferences {
        text += "- " + k + ": " + v + "\n"
    }
    for _, f := range p.Facts {
        text += "- " + f + "\n"
    }
    return text
}
```

## 3. プロンプト組み立て詳細

### 3.1 RecallPack.ToPromptMessages() の実装

```go
func (rp *RecallPack) ToPromptMessages(userMessage string) []llm.Message {
    var messages []llm.Message

    // 1. システムプロンプト（Persona）
    systemPrompt := rp.Persona.SystemPrompt
    if profileText := rp.UserProfile.ToPromptText(); profileText != "" {
        systemPrompt += "\n\n" + profileText
    }
    messages = append(messages, llm.Message{
        Role:    "system",
        Content: systemPrompt,
    })

    // 2. 過去文脈（中期要約 + 長期事実）
    contextText := ""
    if len(rp.MidSummaries) > 0 {
        contextText += "【過去の会話から思い出したこと】\n"
        for _, s := range rp.MidSummaries {
            contextText += "- " + s.Summary + "\n"
        }
    }
    if len(rp.LongFacts) > 0 {
        if contextText == "" {
            contextText += "【過去の会話から思い出したこと】\n"
        }
        for _, f := range rp.LongFacts {
            contextText += "- " + f + "\n"
        }
    }
    if len(rp.KBSnippets) > 0 {
        contextText += "【参考知識】\n"
        for _, kb := range rp.KBSnippets {
            contextText += kb + "\n"
        }
    }
    if contextText != "" {
        messages = append(messages, llm.Message{
            Role:    "system",
            Content: contextText,
        })
    }

    // 3. 直近の会話履歴（ShortContext）
    for _, msg := range rp.ShortContext {
        role := "user"
        if msg.Speaker == SpeakerMio {
            role = "assistant"
        } else if msg.Speaker == SpeakerUser {
            role = "user"
        } else {
            // system, shiro 等はシステムメッセージとして注入
            role = "system"
        }
        messages = append(messages, llm.Message{
            Role:    role,
            Content: msg.Msg,
        })
    }

    // 4. 現在のユーザー入力
    messages = append(messages, llm.Message{
        Role:    "user",
        Content: userMessage,
    })

    return messages
}
```

### 3.2 トークン予算管理

```go
// truncateToTokenBudget はトークン予算内に収まるようにメッセージを削減
// 優先度: システムプロンプト > 直近会話 > ユーザー入力 > 中期要約 > 長期事実 > KB
func (rp *RecallPack) truncateToTokenBudget() {
    // 簡易トークン推定: 日本語1文字 ≈ 1.5トークン、英語1単語 ≈ 1トークン
    // MaxPromptTokens（4000）を超える場合、低優先度から削減
    budget := rp.Constraints.MaxPromptTokens
    if budget == 0 {
        budget = 4000
    }

    // KBSnippets → LongFacts → MidSummaries → ShortContext の順に削減
    estimated := rp.estimateTokens()
    if estimated <= budget {
        return
    }

    // 1. KB削減
    rp.KBSnippets = truncateSlice(rp.KBSnippets, 1)
    if rp.estimateTokens() <= budget {
        return
    }

    // 2. LongFacts削減
    rp.LongFacts = truncateSlice(rp.LongFacts, 1)
    if rp.estimateTokens() <= budget {
        return
    }

    // 3. MidSummaries削減
    rp.MidSummaries = rp.MidSummaries[:min(len(rp.MidSummaries), 1)]
    if rp.estimateTokens() <= budget {
        return
    }

    // 4. ShortContext削減（古いものから削除）
    if len(rp.ShortContext) > 4 {
        rp.ShortContext = rp.ShortContext[len(rp.ShortContext)-4:]
    }
}
```

## 4. インフラ層: ConversationEngine 実装

**ファイル**: `internal/infrastructure/persistence/conversation/engine_impl.go`

```go
package conversation

import (
    "context"
    "fmt"
    "log"
    "time"

    domainconv "github.com/Nyukimin/picoclaw_multiLLM/internal/domain/conversation"
)

// RealConversationEngine は ConversationEngine の実装
type RealConversationEngine struct {
    manager  domainconv.ConversationManager
    persona  domainconv.PersonaState
    profiles map[string]domainconv.UserProfile // インメモリキャッシュ
}

// NewRealConversationEngine は新しい ConversationEngine を作成
func NewRealConversationEngine(
    manager domainconv.ConversationManager,
    persona domainconv.PersonaState,
) *RealConversationEngine {
    return &RealConversationEngine{
        manager:  manager,
        persona:  persona,
        profiles: make(map[string]domainconv.UserProfile),
    }
}

func (e *RealConversationEngine) BeginTurn(ctx context.Context, sessionID string, userMessage string) (*domainconv.RecallPack, error) {
    pack := &domainconv.RecallPack{
        Persona:     e.persona,
        Constraints: domainconv.DefaultConstraints(),
    }

    // 1. Recall（想起）
    recallMessages, err := e.manager.Recall(ctx, sessionID, userMessage, 3)
    if err != nil {
        log.Printf("WARN: Recall failed: %v", err)
        // エラーでも続行（空の RecallPack を返す）
    }

    // 2. Recall 結果を RecallPack に分類
    for _, msg := range recallMessages {
        switch {
        case msg.Speaker == domainconv.SpeakerUser || msg.Speaker == domainconv.SpeakerMio:
            pack.ShortContext = append(pack.ShortContext, msg)
        default:
            // ThreadSummary 等は LongFacts に変換
            pack.LongFacts = append(pack.LongFacts, msg.Msg)
        }
    }

    // 3. UserProfile 読み込み
    if profile, ok := e.profiles[sessionID]; ok {
        pack.UserProfile = profile
    }

    return pack, nil
}

func (e *RealConversationEngine) EndTurn(ctx context.Context, sessionID string, userMessage string, response string) error {
    // 1. ユーザーメッセージを記憶
    userMsg := domainconv.NewMessage(domainconv.SpeakerUser, userMessage, nil)
    if err := e.manager.Store(ctx, sessionID, userMsg); err != nil {
        log.Printf("WARN: Store (user) failed: %v", err)
    }

    // 2. Mio の応答を記憶
    mioMsg := domainconv.NewMessage(domainconv.SpeakerMio, response, nil)
    if err := e.manager.Store(ctx, sessionID, mioMsg); err != nil {
        log.Printf("WARN: Store (mio) failed: %v", err)
    }

    return nil
}

func (e *RealConversationEngine) GetPersona() domainconv.PersonaState {
    return e.persona
}

func (e *RealConversationEngine) GetUserProfile(ctx context.Context, userID string) (domainconv.UserProfile, error) {
    if profile, ok := e.profiles[userID]; ok {
        return profile, nil
    }
    return domainconv.NewUserProfile(userID), nil
}
```

```go
func (m *MioAgent) Chat(ctx context.Context, t task.Task) (string, error) {
    userMessage := t.UserMessage()

    // === v5.1: ConversationEngine による RecallPack 生成 ===
    var messages []llm.Message
    if m.conversationEngine != nil {
        // BeginTurn: Recall + RecallPack 構築
        recallPack, err := m.conversationEngine.BeginTurn(ctx, t.ChatID(), userMessage)
        if err != nil {
            fmt.Printf("WARN: BeginTurn failed: %v\n", err)
        }

        if recallPack != nil && recallPack.HasContext() {
            // RecallPack からプロンプトメッセージを生成
            // （Web検索結果は後で追加）
            messages = recallPack.ToPromptMessages(userMessage)
        }
    }

    // RecallPack がない場合はフォールバック
    if len(messages) == 0 {
        messages = append(messages, llm.Message{Role: "user", Content: userMessage})
    }

    // Web検索（既存ロジック維持）
    if needsSearch && m.toolRunner != nil {
        searchResult, err := m.executeWebSearch(ctx, userMessage)
        if err == nil && searchResult != "" {
            // 検索結果を system メッセージとして挿入
            // messages の先頭（system prompt の後）に挿入
            searchMsg := llm.Message{
                Role:    "system",
                Content: "以下はWeb検索の結果です:\n\n" + searchResult,
            }
            messages = insertAfterSystem(messages, searchMsg)
        }
    }

    req := llm.GenerateRequest{
        Messages:    messages,
        MaxTokens:   512,
        Temperature: 0.7,
    }

    resp, err := m.llmProvider.Generate(ctx, req)
    if err != nil {
        return "", err
    }

    response := resp.Content

    // === v5.1: EndTurn（Store + Thread管理） ===
    if m.conversationEngine != nil {
        if err := m.conversationEngine.EndTurn(ctx, t.ChatID(), userMessage, response); err != nil {
            fmt.Printf("WARN: EndTurn failed: %v\n", err)
        }
    }

    return response, nil
}
```
